The below code is used to render above table with Backbone Model, View and Collection

Student = Backbone.Model.extend({
	defaults: {
		name: "",
		age: 0
	}
});
StudentCollection = Backbone.Collection.extend({
	model: Student
});
studentCollection = new StudentCollection();
student1 = new Student({name: "Testing 1", age: 20 });
student2 = new Student({name: "Testing 2", age: 22 });
studentCollection.add(student1);
studentCollection.add(student2);
Collection = Backbone.View.extend({
	initialize: function(){
		this.render();
	},
	render: function(){
		debugger;
		var table = $("<table border='1'></table>");
		studentCollection.each(function(student){
			var tr = $("<tr></tr>");
			tr.append("<td>" + student.get("name") + "</td>");
			tr.append("<td>" + student.get("age") + "</td>");
			table.append(tr);
		});
		this.$el.html(table);
	}
});	
collection = new Collection({ el: $("#collection") });
