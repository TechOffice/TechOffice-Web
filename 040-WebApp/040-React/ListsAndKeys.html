<div id="container"></div>
<script type="text/babel">
	function NumberList(props){
		const numbers = [1, 2, 3, 4, 5];
		const listItems = numbers.map((number) => 
			<li>{number}</li>
		);
		return (
			<ul>{listItems}</ul>
		);
	}

	ReactDOM.render(
		<NumberList />,
		document.getElementById('container')
	);
</script>
