<div id="container"></div>
<script type="text/babel">
	var data = [{col1: "col 11", col2: "col 12", col3: "col 13"}, 
				{col1: "col 11", col2: "col 12", col3: "col 13"},
				{col1: "col 11", col2: "col 12", col3: "col 13"},
				{col1: "col 11", col2: "col 12", col3: "col 13"},
				{col1: "col 11", col2: "col 12", col3: "col 13"},
				{col1: "col 11", col2: "col 12", col3: "col 13"},
				{col1: "col 11", col2: "col 12", col3: "col 13"},
				{col1: "col 11", col2: "col 12", col3: "col 13"}];

	// Paging Table 
	class Table extends React.Component {
		constructor(props){
			super(props);
			this.data = data
			this.columns = ['col1', 'col2', 'col3'];
			this.header = ['Column 1', 'Column 2', 'Column 3'];
		}
		render(){
			var me = this;
			var rows = [];
			me.data.forEach(function(row){
				var cols = [];
				me.columns.forEach(function(column){
					cols.push(<td>{row[column]}</td>);
				});
				rows.push(<tr>{cols}</tr>)
			});
			return <table>{rows}</table>;
		}
	};

	ReactDOM.render(
		<Table />,
		document.getElementById('container')
	);
</script>
