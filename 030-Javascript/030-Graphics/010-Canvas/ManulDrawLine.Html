Canvas

Canvas is used to draw graphics on the fly via JavaScript. The 2d object of this would provide method to draw text, line, box, circles and more on it.

Draw Line Example

This is an example of canvas which allow you use mouse to draw line on it.

Main Part of the Script

canvas = document.getElementById("canvas");
context = canvas.getContext("2d");
context.beginPath();
context.moveTo(startX, startY);
context.lineTo(endX, endY);
context.stroke();

Complete Script

document.onreadystatechange = function() {
    if (document.readyState == 'complete'){
        var canvas = document.getElementById("canvas");
        var ctx = canvas.getContext("2d");

        canvas.flag = false;
        canvas.startX = 0;
        canvas.startY = 0;
        canvas.endX = 0;
        canvas.endY = 0;
        canvas.objWidth = 0;
        canvas.objHeight = 0;
        canvas.mousedown = function(event){
            this.flag = true;
            this.startX = event.clientX;
            this.startY = event.clientY;
        };
        canvas.addEventListener("mousedown", canvas.mousedown);
        canvas.mouseup = function(event){
            this.flag = false;
        };
        canvas.addEventListener("mouseup", canvas.mouseup);
        canvas.mousemove = function(event){
            if (!this.flag){
                return;
            }
            this.endX = event.clientX;
            this.endY = event.clientY;
            ctx.fillStyle = "#FF0000";
            ctx.beginPath();
            ctx.moveTo(this.startX, this.startY);
            ctx.lineTo(this.endX, this.endY);
            ctx.stroke();
            this.startX = this.endX;
            this.startY = this.endY;
        };
        canvas.addEventListener("mousemove", canvas.mousemove);
        canvas.mouseleave = function(event){
            
        };
        canvas.addEventListener("mouseleave", canvas.mouseleave);
        
    }
};
